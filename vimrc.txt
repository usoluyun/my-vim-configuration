"Key mapping
map <silent> <C-F2> :if &guioptions =~# 'T' <Bar>
                                          \set guioptions-=T  <Bar>
                                           \set guioptions-=m  <Bar>
				            \set guioptions-=r <Bar>
                                      \else <Bar>
                                          \set guioptions+=T  <Bar>
                                           \set guioptions+=m  <Bar>
					    \set guioptions+=r <Bar>
                                       \endif<CR>
